# useFilterState

A custom, composeable hook for handling filter state.

```ts
// Syntax:
const filterState = useFilterState<T>(args: IFilterStateArgs<T>);

// Arguments:
interface IFilterStateArgs<T> {
  items: T[];
  filterCategories: T[]; // Defaults to []
}

// Return value:
interface IFilterState<T> {
  filteredItems: T[];
  filterValues: any;
  setFilterValues: any;
}
```
